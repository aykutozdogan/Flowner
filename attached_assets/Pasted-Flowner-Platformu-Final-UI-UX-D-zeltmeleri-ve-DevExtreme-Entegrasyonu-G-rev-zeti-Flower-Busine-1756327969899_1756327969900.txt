Theme Switching Logic
typescript// useTheme hook
const themes = [
  { name: "Light", css: "dx.light.css" },
  { name: "Dark", css: "dx.dark.css" },
  { name: "Material Light", css: "dx.material.blue.light.css" },
  { name: "Material Dark", css: "dx.material.blue.dark.css" }
];

const switchTheme = (theme) => {
  document.querySelector('link[data-theme]').href = theme.css;
  localStorage.setItem('theme', theme.name);
};
